        }

    {
        public Image Img { get; set; }

        {
            return Img;
        }
    }

创建可移动的类Movething

//朝向
    enum Direction
    {
        Up,
        Down,
        Left,
        Right
    }
    class Movething :GameObject
    {
        //四个方向
        public Bitmap BitmapUp { get; set; }
        public Bitmap BitmapDown { get; set; }
        public Bitmap BitmapLeft { get; set; }
        public Bitmap BitmapRight { get; set; }

        //移动速度
        public int Speed { get; set; }

        public Direction Dir { get; set; }

        protected override Image GetImage()
        {
            Bitmap bitmap = null;
            switch (Dir)
            {
                case Direction.Up:
                    bitmap = BitmapUp;
                    break;
                case Direction.Down:
                    bitmap = BitmapDown;
                    break;
                case Direction.Left:
                    bitmap = BitmapLeft;
                    break;
                case Direction.Right:
                    bitmap = BitmapRight;
                    break;
                default:
                    break;
            }
            //设置透明度
            bitmap.MakeTransparent(Color.Black);
            return bitmap;
        }
    }

9.4 控制我的坦克不要穿墙——碰撞检测

系统提供的类Rectangle

  • 存储一组整数，共四个，表示一个矩形的位置和大小。
  • 对于一个矩形，我们只需要知道它左上角的位置，以及它的宽高即可确定

任意一个游戏物体都存在一个矩形

在GameObject类中添加一个新的内容GetRectangle()方法，用于返回一个Rectangle对象的引用

public Rectangle GetRectangle()
        {
            Rectangle rectangle = new Rectangle(X, Y, Width, Height);
            return rectangle;
        }

构造函数

Rectangle(Int32, Int32, Int32, Int32) 用指定的位置和大小初始化 Rectangle 类的新实例。
Rectangle(Point, Size) 用指定的位置和大小初始化 Rectangle 类的新实例。

然后我们可以把碰撞检测的代码放到GameObjectManager中，定义如下

public static NotMovething IsCollidedWall(Rectangle rt)

由于子弹碰到墙我们需要知道是那堵墙，所以我们把返回值定义为NotMovething，如果返回值为null则说明没有发生碰撞

Rectangle.IntersectsWith(Rectangle) 方法

  • 命名空间:

    System.Drawing

  • 程序集:

    System.Drawing.dll

确定此矩形是否与 rect 相交。

先写关于红墙的碰撞检测代码如下

public static NotMovething IsCollidedWall(Rectangle rt)
        {
            //遍历所有的墙
            foreach (var item in _wallList)
            {
                //判断是否与传递过来的矩形发生碰撞
                if (item.GetRectangle().IntersectsWith(rt))
                {
                    return item;
                }
            }
            return null;

class EnemyTank:Movething
    {
        Random _r = new Random();
        public EnemyTank(int x, int y, int speed,Bitmap bmpDown, Bitmap bmpUp, Bitmap bmpRight, Bitmap bmpLeft)
        {
            //初始坐标
            X = x;
            Y = y;
            //初始速度
            Speed = speed;
            //四个方向的图片
            BitmapDown = bmpDown;
            BitmapUp = bmpUp;
            BitmapRight = bmpRight;
            BitmapLeft = bmpLeft;
            //初始方向
            Dir = Direction.Down;
        }
        private void _ChangeDirection()
        {
            while (true)
            {
                Direction dir = (Direction)_r.Next(0, 4);  //Direction为0-3
                if(dir == Dir)
                {
                    continue;
                }
                else
                {
                    Dir = dir;
                    break;
                }
            }
            //更改方向后仍然可能存在障碍物,所以我们需要再执行一次_MoveCheck
            _MoveCheck();

        }
        public override void Update()
        {
            _MoveCheck();   //移动检查
            _Move();
            base.Update();
        }

        private void _MoveCheck()
        {
            #region 检查有没有超出窗体边界，判断四个方向即可
            if (Dir == Direction.Up)
            {
                if (Y - Speed < 0)
                {
                    _ChangeDirection();
                    return;
                }
            }
            else if (Dir == Direction.Down)
            {
                if (Y + Speed + Height > 450)
                {
                    _ChangeDirection();
                    return;
                }
            }
            else if (Dir == Direction.Left)
            {
                if (X - Speed < 0)
                {
                    _ChangeDirection();
                    return;
                }
            }
            else
            {
                if (X + Speed + Height > 450)
                {
                    _ChangeDirection();
                    return;
                }
            }
            #endregion
            //检查有没有和其他元素发生碰撞

            Rectangle rectangle = GetRectangle();

            switch (Dir)
            {
                case Direction.Up:
                    rectangle.Y -= Speed;
                    break;
                case Direction.Down:
                    rectangle.Y += Speed;
                    break;
                case Direction.Left:
                    rectangle.X -= Speed;
                    break;
                case Direction.Right:
                    rectangle.X += Speed;
                    break;
            }

            if (GameObjectManager.IsCollidedWall(rectangle) != null)
            {
                _ChangeDirection();
                return;
            }
            if (GameObjectManager.IsCollidedSteel(rectangle) != null)
            {
                _ChangeDirection();
                return;
            }
            if (GameObjectManager.IsCollidedBoss(rectangle) != false)
            {
                _ChangeDirection();
                return;
            }
        }

        private void _Move()
        {
            switch (Dir)
            {
                case Direction.Up:
                    Y -= Speed;
                    break;
                case Direction.Down:
                    Y += Speed;
                    break;
                case Direction.Left:
                    X -= Speed;
                    break;
                case Direction.Right:
                    X += Speed;
                    break;
            }
        }
    }

          _changeDirCount = 0;
      }

同时在构造方法中添加敌人改变所需要的时间，目前设置为80帧

13.添加音效（游戏结束）

一共存在五种音效

  1. hit 敌人坦克打到我的坦克的音效
  2. add 敌人生成的音效
  3. start 游戏开始的音效
  4. fire 子弹发射的音效
  5. blast 爆炸

添加一个SoundManager类用于存放音效

class SoundManager
    {
        private static SoundPlayer _startPlayer = new SoundPlayer();
        private static SoundPlayer _addPlayer = new SoundPlayer();
        private static SoundPlayer _blastPlayer = new SoundPlayer();
        private static SoundPlayer _firePlayer = new SoundPlayer();
        private static SoundPlayer _hitPlayer = new SoundPlayer();
        public static void InitSound()
        {
            _startPlayer.Stream= Resources.start;
            _addPlayer.Stream = Resources.add;
            _blastPlayer.Stream = Resources.blast;
            _firePlayer.Stream = Resources.fire;
            _hitPlayer.Stream = Resources.hit;
        }

        //游戏开始的音效，只需要在GameFramework中添加一次
        public static void PlayStart()
        {
            _startPlayer.Play();
        }
        //增加一个敌人的音效，在GameOjectManager中的EnemyBorn
        public static void PlayAdd()
        {
            _addPlayer.Play();
        }
    	//爆炸音效，比如子弹攻击到红墙、BOOS或者我的坦克攻击到敌人坦克
        public static void PlayBlast()
        {
            _blastPlayer.Play();
        }
    	//子弹发射的音效
        public static void PlayFire()
        {
            _firePlayer.Play();
        }
    	//敌人坦克打到我的坦克的音效
        public static void PlayHit()
        {
            _hitPlayer.Play();
        }
    }

将InitSound和PlayStart方法添加到GameFramework中的Start方法中

因为这两个方法都只需要在游戏开始时运行一次

然后将PlayAdd添加到GameObjectManager类中的_EnemyBurn方法中

将PlayBlast方法添加到Bullet类的_MoveCheck方法中

  1. 红墙碰到子弹会爆炸
  2. BOOS碰到子弹会爆炸GameOver
  3. 我的坦克发射的子弹碰到敌人坦克，敌人坦克会爆炸

然后敌人子弹碰到我的坦克会产生Hit音效

我的坦克发射子弹会产生Fire音效

到这整个游戏的开发就结束了，存在许多可以优化的问题，比如说音效无法同时生效，然后我因为将坦克的速度设置的大了点，导致产生空气墙的问题（原理是提前进行碰撞测试，因为速度过大，导致会产生空气墙），解决方案是将坦克速度设置的小一点
