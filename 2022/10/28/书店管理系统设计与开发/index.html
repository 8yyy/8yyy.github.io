7. 系统加载界面功能实现

主要使用Timer控件实现进度条的加载

public partial class Splash : Form
{
    public Splash()
    {
        InitializeComponent();
    }
    int startPos = 0;
    private void timer1_Tick(object sender, EventArgs e)
    {
        startPos++;
        Myprogress.Value = startPos;
        PercentageLbl.Text = startPos.ToString()+"%";
        if (Myprogress.Value == 100)
        {
            Myprogress.Value = 0;
            timer1.Stop();
            Login log = new Login();
            log.Show();
            this.Hide();
        }
    }

    private void Splash_Load(object sender, EventArgs e)
    {
        timer1.Start();
    }
}

SqlConnection Con = new SqlConnection(@"Data Source=.;Initial Catalog=WhiteBookShopDb;Integrated Security=True;Pooling=False");

8.1 保存功能

保存功能如下

private void SaveBtn_Click(object sender, EventArgs e)
{
    if(BTitleTb.Text=="" || BautTb.Text=="" || QtyTb.Text=="" || PriceTb.Text=="" || BCatCb.SelectedIndex == -1)
    {
        MessageBox.Show("信息缺失！！！");
    }
    else
    {
        try
        {
            Con.Open();
            string query = $"insert into BookTb1 values('{BTitleTb.Text}','{BautTb.Text}','{BCatCb.SelectedItem}','{QtyTb.Text}','{PriceTb.Text}')";
            SqlCommand cmd = new SqlCommand(query, Con);
            cmd.ExecuteNonQuery();
            MessageBox.Show("书籍信息保存成功");
            Con.Close();
            //这两个方法是后面加的
            Populate();
            Reset();
        }
        catch (Exception ex)
        {
            MessageBox.Show(ex.Message);
        }
    }
}

8.2 设置数据表格控件

编写一个函数用于显示datagirdview数据

//显示datagirdview数据
private void Populate()
{
    Con.Open();
    string query = "select * from BookTb1";
    SqlDataAdapter sda = new SqlDataAdapter(query, Con);
    SqlCommandBuilder builder = new SqlCommandBuilder(sda);
    var ds = new DataSet();
    sda.Fill(ds);
    BookGDV.DataSource = ds.Tables[0];
    Con.Close();
}

同时将上述方法添加到Books类中的构造方法中，以运行时直接显示表格的数据

对数据表格控件BookGDV进行修改如下，以显示更好的效果，

将AutoSizeColumnsMode设置为Fill，调整列宽，使所有列的宽度正好填充控件的显示区域，只需要水平滚动保证列宽在

将RowHeaderVisible设置为False，不显示行号

将SelectionMode设置为FullRowSelect，整行选定

将ReadOnly设置为Ture，表示表格只读

将CellBorderStyle设置为SingleHorizontal，单元格划线使用水平单线边框

将GirdColor设置GrayText，灰色

设置RowsDefaultCellStyle中的外观SelectionBackColor，设置选中行标题颜色

编写一个方法进行过滤

//根据书目类型进行过滤
private void Filter()
{
    Con.Open();
    string query = $"select * from BookTb1 where BCat ='{CatCbSearch.SelectedItem.ToString()}'";
    SqlDataAdapter sda = new SqlDataAdapter(query, Con);
    SqlCommandBuilder builder = new SqlCommandBuilder(sda);
    var ds = new DataSet();
    sda.Fill(ds);
    BookGDV.DataSource = ds.Tables[0];
    Con.Close();
}
private void CatCbSearch_SelectionChangeCommitted(object sender, EventArgs e)

private void DeleteBtn_Click(object sender, EventArgs e)
{
    if (key==0)
    {
        MessageBox.Show("信息缺失！！！");

    else
    {
        try
        {
